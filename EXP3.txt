SELECT A.*,B.S
FROM STUDENTNEW A, SEMSEC B, CLASS C
WHERE A.USN=C.USN AND B.SSID=C.SSID AND B.SEM=4 AND B.S='C';




SELECT SEM,S,GENDER,COUNT(*)
FROM STUDENTNEW S, SEMSEC S1, CLASS C
WHERE S.USN=C.USN AND S1.SSID=C.SSID
GROUP BY(GENDER,SEM,S)
ORDER BY(SEM);

SELECT USN,FINALIA,
CASE
WHEN FINALIA BETWEEN 45 AND 50 THEN 'OUTSTANDING'
WHEN FINALIA BETWEEN 40 AND 45 THEN 'GOOD'
WHEN FINALIA BETWEEN 30 AND 40 THEN 'AVERAGE'
WHEN FINALIA <30 THEN 'WEAK'
END
AS CATEGORY
FROM IAMARKS I,SEMSEC S1
WHERE I.SSID=S1.SSID AND SEM=8 AND S IN('A','B','C');
